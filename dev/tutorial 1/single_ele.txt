units metal
dimension 3
boundary p p p
atom_style atomic

lattice ${lattice_type} ${lattice_constant}
region box block 0 1 0 1 0 1 units lattice
create_box 1 box
create_atoms 1 box replicate ${replicate_x} ${replicate_y} ${replicate_z}

pair_style eam/alloy
pair_coeff * * ${potential_file} ${element_symbol}

compute eng all pe/atom
compute eatoms all reduce sum c_eng

# ------- Let box size change under zero pressure to find Lattice Constant -------- #
reset_timestep 0
fix 1 all box/relax iso 0.0 vmax 0.001
thermo 10
thermo_style custom step pe lx ly lz press pxx pyy pzz c_eatoms
min_style cg
minimize 1e-25 1e-25 5000 10000

variable natoms equal "count(all)"
variable teng equal "c_eatoms"
variable length equal "lx"
variable ecoh equal "v_teng/v_natoms"

print "Total energy (eV) = ${teng};"
print "Number of atoms = ${natoms};"
print "Lattice constant (Angstroms) = ${length};"
print "Cohesive energy (eV) = ${ecoh};"
print "All done!"
